local function displayAnnouncement(player, text)
    local billboardGui = Instance.new("BillboardGui")
    billboardGui.Size = UDim2.new(0, 200, 0, 50)
    billboardGui.StudsOffset = Vector3.new(0, 3, 0)
    billboardGui.AlwaysOnTop = true
    billboardGui.Parent = player.Character and player.Character:FindFirstChild("Head")

    local textLabel = Instance.new("TextLabel")
    textLabel.Size = UDim2.new(1, 0, 1, 0)
    textLabel.BackgroundTransparency = 1
    textLabel.TextColor3 = Color3.fromRGB(255, 0, 0)
    textLabel.TextScaled = true
    textLabel.Text = text
    textLabel.Parent = billboardGui
end

local whitelist = {

}


local function isWhitelisted(username)
    for _, v in ipairs(whitelist) do
        if username == v then
            return true
        end
    end
    return false
end

-- 显示 UI 界面和加载脚本
local function showUI()
    if not isWhitelisted(game.Players.LocalPlayer.Name) then
        -- 如果玩家不在白名单中，显示无法使用脚本的提示
        local screenGui = Instance.new("ScreenGui")
        local frame = Instance.new("Frame")
        local textLabel = Instance.new("TextLabel")

        screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

        frame.Parent = screenGui
        frame.Size = UDim2.new(0, 400, 0, 200)
        frame.Position = UDim2.new(0.5, -200, 0.5, -100)
        frame.BackgroundColor3 = Color3.fromRGB(255, 0, 0)

        local uiCorner = Instance.new("UICorner")
        uiCorner.Parent = frame
        uiCorner.CornerRadius = UDim.new(0.1, 0)

        textLabel.Parent = frame
        textLabel.Size = UDim2.new(1, 0, 1, 0)
        textLabel.Text = "您不在白名单中，无法使用此脚本"
        textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        textLabel.Font = Enum.Font.GothamBold
        textLabel.TextSize = 30
        textLabel.BackgroundTransparency = 1

        return -- 终止脚本执行
    else
        -- 如果玩家在白名单中，显示成功提示
        local screenGui = Instance.new("ScreenGui")
        local frame = Instance.new("Frame")
        local textLabel = Instance.new("TextLabel")

        screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

        frame.Parent = screenGui
        frame.Size = UDim2.new(0, 400, 0, 200)
        frame.Position = UDim2.new(0.5, -200, 0.5, -100)
        frame.BackgroundColor3 = Color3.fromRGB(0, 170, 0)

        local uiCorner = Instance.new("UICorner")
        uiCorner.Parent = frame
        uiCorner.CornerRadius = UDim.new(0.1, 0)

        textLabel.Parent = frame
        textLabel.Size = UDim2.new(1, 0, 1, 0)
        textLabel.Text = "你已成功添加白名单！"
        textLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
        textLabel.Font = Enum.Font.GothamBold
        textLabel.TextSize = 30
        textLabel.BackgroundTransparency = 1

        wait(3) -- 等待3秒
        screenGui:Destroy()

        
        loadstring(game:HttpGet('https://raw.githubusercontent.com/ToraScript/Script/main/RideACart'))()
    end
end


showUI()
